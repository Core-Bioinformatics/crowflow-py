Integrating Crow and OpenAI Models for Reliable Clustering and Label Generation

Crow is a Python package designed to evaluate the stability of clustering algorithms through repeated stochastic clustering. It uses Element-Centric Similarity (ECS) and Element-Centric Consistency (ECC) to assess the consistency of clustering results across multiple runs.

This notebook demonstrates how to integrate cutting-edge models from OpenAI with Crow to achieve reliable clustering results, extract meaningful insights from the identified clusters, and generate informative labels based on these insights.

def plot_umap(embedding_df, color_by=None, color_label=None):
    plt.figure(figsize=(8, 6))

    if color_by is not None:
        groups = embedding_df.groupby(color_by)
        # Choose a categorical colormap with enough distinct colors
        cmap = plt.get_cmap('tab20', len(groups))  # 'tab20' offers up to 20 distinct colors
        
        for i, (name, group) in enumerate(groups):
            plt.scatter(
                group['UMAP_1'],
                group['UMAP_2'],
                s=5,
                alpha=0.7,
                color=cmap(i),
                label=name
            )
        plt.legend(title=color_label if color_label else color_by, bbox_to_anchor=(1.05, 1), loc='upper left')
    else:
        plt.scatter(
            embedding_df['UMAP_1'],
            embedding_df['UMAP_2'],
            s=5,
            alpha=0.7
        )

    plt.title("UMAP Projection of Embeddings")
    plt.xlabel("UMAP 1")
    plt.ylabel("UMAP 2")
    plt.tight_layout()
    plt.show()

# Define the OpenAI completions model and API key
COMPLETIONS_MODEL = "gpt-4o-mini-2024-07-18"

# Function to generate a prompt for analyzing clusters
def generate_prompt(cluster_id, sample_reviews):
    prompt = f"""
    Cluster {cluster_id} Reviews:
    {sample_reviews}

    --Instructions--
    # Theme: Identify the main product or topic.
    # Sentiment: State whether the reviews are predominantly positive, negative, or mixed.
    # Variability: List succinctly all the different items in this cluster.
    # Patterns: List any repeated words, phrases, or issues.
    # Label: Suggest a short label for this cluster.
    """
    return prompt

# Function to analyze a cluster using OpenAI's completions model
def analyze_cluster(cluster_id, df_cluster, model=COMPLETIONS_MODEL):
    sample_reviews = "\n".join(df_cluster['Summary'].tolist())
    prompt = generate_prompt(cluster_id, sample_reviews)

    response = client.chat.completions.create(
        messages=[
            {
                "role": "user",
                "content": prompt,
            }
        ],
        model=model,
        max_tokens=500,
        temperature=0.7
    )
    return response.choices[0].message.content

def print_description_of_clusters(df, labels):
    cluster_analysis_dict = {}

    # Iterate through clusters and analyze them
    for cluster in sorted(df[labels].unique()):
        df_cluster = df[df[labels] == cluster]
        print(f"Cluster {cluster}:")
        cluster_analysis_dict[f'cluster_{cluster}'] = analyze_cluster(cluster, df_cluster)
        print(cluster_analysis_dict[f'cluster_{cluster}'])
        print("-" * 100)
        print()
